```js
let prefix = null

function genOptions () {
  prefix = Math.random().toString(36).slice(2, 5)
  return Array.apply(null, { length: 200 }).map((v, i) => ({
    label: prefix + 'Option ' + i,
    value: prefix + i,
    disabled: i % 5 === 0
  }))
}

function genValues () {
  return Array.apply(null, { length: 100 }).map((v, i) => prefix + i)
}

export default {
  data () {
    return {
      options: genOptions(),
      value: genValues()
    }
  },
  mounted () {},
  methods: {
    regenOptions () {
      this.options = genOptions()
    },
    regenValues () {
      this.value = genValues()
    }
  }
}
```
